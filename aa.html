#数据处理
cate=df["试卷/场次"].tolist()
data1=df["上午场"].tolist()
data2=df["下午场"].tolist()

#柱状图制作
bar = (Bar()
    .add_xaxis(cate)
    .add_yaxis('基础知识成绩', data1, gap="0%",category_gap='50%')
    .add_yaxis('应用技术成绩', data2, gap="0%",category_gap='50%')
    .set_global_opts(title_opts=opts.TitleOpts(title="软中真题模拟成绩", subtitle=""),toolbox_opts=opts.ToolboxOpts())
    .set_series_opts(
        label_opts=opts.LabelOpts(is_show=True),
        markline_opts=opts.MarkLineOpts(
            data=[
                opts.MarkLineItem(type_="average", name="平均值"),
            ]
        ),
    ) 
   )
bar.render_notebook()
